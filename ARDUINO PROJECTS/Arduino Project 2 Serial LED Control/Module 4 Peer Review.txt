/*
 * Arduino sketch to control the built-in LED (pin 13) 
 * via serial monitor commands ('1' for ON, '0' for OFF).
 */

const int ledPin = 13; // The pin number for the built-in LED
char incomingChar;     // Variable to store the character received from the serial port

void setup() {
  // Initialize the LED pin as an output pin
  pinMode(ledPin, OUTPUT); 
  // Start serial communication at 9600 bits per second (baud rate)
  Serial.begin(9600);
  // Ensure the LED is initially off
  digitalWrite(ledPin, LOW);
}

void loop() {
  // Check if there is data available to read from the serial port
  if (Serial.available() > 0) {
    // Read the incoming character and store it
    incomingChar = Serial.read();

    // If the received character is '1', turn the LED ON (HIGH)
    if (incomingChar == '1') {
      digitalWrite(ledPin, HIGH);
    }
    // If the received character is '0', turn the LED OFF (LOW)
    else if (incomingChar == '0') {
      digitalWrite(ledPin, LOW);
    }
    // Note: Any other character received is ignored.
  }
}
